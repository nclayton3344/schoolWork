CREATE TABLE COLLECTOR (
COLLECTOR_ID INTEGER IDENTITY PRIMARY KEY,
COLLECTOR_NAME VARCHAR(25) NOT NULL,
COLLECTOR_AGE INTEGER NOT NULL
);

CREATE TABLE TEAM (
TEAM_ID INTEGER IDENTITY PRIMARY KEY,
TEAM_NAME VARCHAR(15) NOT NULL,
TEAM_ABBREV VARCHAR(3) NOT NULL,
TEAM_CITY VARCHAR(15) NOT NULL
);

CREATE TABLE PLAYER (
PLAYER_ID INTEGER IDENTITY PRIMARY KEY,
TEAM_ID INTEGER NOT NULL,
PLAYER_NAME VARCHAR(20) NOT NULL,
PLAYER_HEIGHT INT NOT NULL,
PLAYER_WEIGHT INT NOT NULL,
PLAYER_POSITION VARCHAR(3) NOT NULL,
PLAYER_NUMBER INT NOT NULL,
PLAYER_ROOKIE CHAR(1) NOT NULL
FOREIGN KEY (TEAM_ID) REFERENCES TEAM (TEAM_ID) ON DELETE CASCADE
);

CREATE TABLE COLLECTIONS (
COLLECTION_ID INTEGER IDENTITY PRIMARY KEY,
COLLECTOR_ID INTEGER NOT NULL,
COLLECTION_NAME VARCHAR(20) NOT NULL,
COLLECTION_DESC VARCHAR(25) NOT NULL,
COLLECTION_CARDNUM INT NOT NULL
FOREIGN KEY (COLLECTOR_ID) REFERENCES COLLECTOR (COLLECTOR_ID) ON DELETE CASCADE
);

CREATE TABLE CARDS (
CARD_ID INTEGER IDENTITY PRIMARY KEY,
COLLECTION_ID INTEGER NOT NULL,
PLAYER_ID INTEGER NOT NULL,
DATE_CREATED DATE NOT NULL,
CARD_NAME VARCHAR(20) NOT NULL,
CARD_DESC VARCHAR(20) NOT NULL,
FOREIGN KEY (COLLECTION_ID) REFERENCES COLLECTIONS (COLLECTION_ID) ON DELETE CASCADE,
FOREIGN KEY (PLAYER_ID) REFERENCES PLAYER (PLAYER_ID) ON DELETE CASCADE
);
